    <script>
        window.addEventListener('load', function () {
            if (window.Telegram && window.Telegram.WebApp) {
                const tg = window.Telegram.WebApp;
                tg.expand();

                tg.MainButton.setText("Кликни для теста");
                tg.MainButton.setParams({
                    color: "#00ff88",
                    text_color: "#000000"
                });
                tg.MainButton.show();

                tg.MainButton.offClick();  // снимаем старые обработчики

                tg.MainButton.onClick(function () {
                    tg.showAlert("Main Button нажата! Работает без дублей.");
                });
            } else {
                document.body.innerHTML += '<p style="color:red;">Запусти меня внутри Telegram!</p>';
            }
        });
    </script>
